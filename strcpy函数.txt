char * strcpy(char *dest, char *src)
{
	if (dest == NULL || src == NULL)
	{
		return NULL;
	}
 
	char *pTemp = dest;
 
	while ((*dest++ = *src++) != '\0')
	{
		//空函数体
	}
 
	return pTemp;
}
 
void main
{
	char src[] = "this is a test";
	char *dest = new char[strlen(src) + 1];
 
	strcpy(dest, src);
}
